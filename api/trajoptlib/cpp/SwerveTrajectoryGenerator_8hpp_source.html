173class TRAJOPT_DLLEXPORT SwerveTrajectory {
174 public:
176 std::vector<SwerveTrajectorySample> samples;
177
178 SwerveTrajectory() = default;
179
185 explicit SwerveTrajectory(std::vector<SwerveTrajectorySample> samples)
186 : samples{std::move(samples)} {}
187
193 explicit SwerveTrajectory(const SwerveSolution& solution) {
194 double ts = 0.0;
195 for (size_t sample = 0; sample < solution.x.size(); ++sample) {
196 samples.emplace_back(
197 ts, solution.x[sample], solution.y[sample],
198 std::atan2(solution.thetasin[sample], solution.thetacos[sample]),
199 solution.vx[sample], solution.vy[sample], solution.omega[sample],
200 solution.ax[sample], solution.ay[sample], solution.alpha[sample],
201 solution.moduleFx[sample], solution.moduleFy[sample]);
202 ts += solution.dt[sample];
203 }
204 }
205};
